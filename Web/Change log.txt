CREATE PROCEDURE [dbo].[spGet_MPGSBinCodeBankList]
	@BinCode        VARCHAR(15) = NULL

AS 
BEGIN
	SET NOCOUNT ON
	Declare @Bin varchar(15)
	Declare @BankCode varchar(20)
	Declare @BankName varchar(200)

select @Bin = bin.Bin,@BankCode = bankcode,@BankName = BankName  from MPGSBinList bin
left join MPGSBankList bank on bin.BankBinListId = bank.id
where bin.bin = @BinCode

if(@Bin is not null)
BEGIN 
select @Bin as Bin,@BankCode as BankCode,@BankName as BankName
END
ELSE
BEGIN
select '' as Bin,'ELX-X' as BankCode,'Other Bank' as BankName
END

END
